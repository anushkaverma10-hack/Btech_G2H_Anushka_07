#include <stdio.h>

int main() {
    int dec, bin[32], i = 0, j;

    printf("Enter decimal number: ");
    scanf("%d", &dec);

    while (dec > 0) {
        bin[i] = dec % 2;
        dec = dec / 2;
        i++;
    }

    printf("Binary = ");
    for (j = i - 1; j >= 0; j--) {
        printf("%d", bin[j]);
    }

    return 0;
